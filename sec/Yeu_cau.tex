\section{IV. Phân Tích Yêu Cầu Hệ Thống}

\begin{frame}
\frametitle{Yêu Cầu Chức Năng \& Phi Chức Năng}
\begin{columns}
    \column{0.50\textwidth}
    \textbf{1. Yêu Cầu Chức Năng}
    \begin{itemize}
        \item \textbf{Gameplay:} Tạo, tham gia phòng, chia bài, phân vai, logic đặt thẻ đường, hành động, xác định thắng-thua.
        \item \textbf{Xã hội - Tương tác:} Điều khiển nhân vật 3D, nhạc cụ, thay đổi trang phục, Voice-Text Chat, Emotes.
    \end{itemize}

    \column{0.50\textwidth}
    \textbf{2. Yêu Cầu Phi Chức Năng}
    \begin{itemize}
        \item Hiệu năng: \textbf{ 60 FPS}, tải màn < 10 giây
        \item Độ trễ: dưới \textbf{100ms} cho các thao tác chính
        \item Tin cậy: Xử lý disconnect không làm sập phòng
        \item Mở rộng: Kiến trúc mô-đun hóa
    \end{itemize}
\end{columns}

\end{frame}

% SLIDE 7: Yêu Cầu Kỹ Thuật \& Trải Nghiệm Người Dùng
\begin{frame}
\frametitle{Yêu Cầu Kỹ Thuật \& Trải Nghiệm Người Dùng}

\begin{columns}
    \column{0.5\textwidth}
    \textbf{3. Yêu Cầu Kỹ Thuật}
    \begin{itemize}
        \item Nền tảng: Windows 10/11, Chuột + Bàn phím
        \item Game Engine: Hỗ trợ 3D \& Multiplayer
        \item Networking: Client-Host (NGO)
        \item Đồ họa: Low-poly / Stylized
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{4. Yêu Cầu Trải Nghiệm Người Dùng (UX)}
    \begin{itemize}
        \item Giao diện rõ ràng, thao tác mượt mà
        \item Phản hồi nhanh (âm thanh, hiệu ứng)
        \item Môi trường vui vẻ, nhạc nền nhẹ nhàng
        \item Hướng dẫn \& Tooltip dễ tiếp cận
    \end{itemize}
\end{columns}
\end{frame}
